Formatting Time

var time_data = [
  {"elapsed": "00:00:15", "value": 9},
  {"elapsed": "00:00:17.56", "value": 13},
  {"elapsed": "00:01:20", "value": 10},
  {"elapsed": "00:00:10", "value": 20},
  {"elapsed": "00:01", "value": 15}
];
Morris.Line({
  element: 'graph',
  data: time_data,
  xkey: 'elapsed',
  ykeys: ['value'],
  labels: ['value'],
  parseSeconds: true
});
